--Declare @IdCab Int

ALTER PROCEDURE [dbo].[spfe_BuscaIdFacturacion]
	@Empresa VarChar(11),
	@TipoDoc Varchar(3),
    @Serie   Varchar(4),
    @Numero  Varchar(10)
	
AS

   Select EmpresaRuc,IdFacturacion
     From FE_Cabecera 
    Where EmpresaRUC=@Empresa 
      And ComprobanteTipo=@TipoDoc 
      And ComprobanteSerie=@Serie 
      And ComprobanteNumero=@Numero
